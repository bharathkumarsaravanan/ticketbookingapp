oldWomens.map((woman, index) => {
    if(index < rCount){
        let tempValue = index * 6;
        axios.post('http://localhost:9000/home/agent/'+user+'/booking/'+woman.id,{seat:tempValue})
        .then(response => setPassengers(response.data.return))
        setSelected(prev => [...prev, tempValue])
    }
    else{
        if(index < rCount+rCount){
            let tempValue = ((index-rCount) *6) + 5;
            axios.post('http://localhost:9000/home/agent/'+user+'/booking/'+woman.id,{seat:tempValue})
            .then(response => setPassengers(response.data.return))
            setSelected(prev => [...prev, tempValue])
        }
        else{
            axios.post('http://localhost:9000/home/agent/'+user+'/booking/'+woman.id,{seat:"waiting.."})
            .then(response => setPassengers(response.data.return))
        }

    }

})
